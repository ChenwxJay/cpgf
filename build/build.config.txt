set(BUILD_DEBUG
	0
)

# set 1 to enable Lua binding, 0 to disable
set(HAS_LUA
	1
)

# set 1 to enable V8 binding, 0 to disable
set(HAS_V8
	1
)

# set 1 to enable C++ 0X in GCC, then rvalue reference is enabled
# 0 to disable rvalue reference
set(SUPPORT_GCC_0X
	1
)

# whether we have GLUT library
set(HAS_GLUT
	1
)

# To enable unit test, change the below folder to UnitTest++.
set(UNIT_TEST_DIR
	/source/UnitTest++
)

set(TEST_GCOV 0)
if(UNIX)
	# disable gcov on Linux, otherwise, valgrind will become super slow.
	set(TEST_GCOV 0)
endif(UNIX)

### Lua config

set(LUA_DIR
	/source/lua
)

set(LUA_INC ${LUA_DIR}/src/ ${LUA_DIR}/etc/)

if(CMAKE_COMPILER_IS_GNUCC)
	set(LUA_LIB ${LUA_DIR}/src/liblua.a)
endif(CMAKE_COMPILER_IS_GNUCC)

if(MSVC)
	set(LUA_LIB ${LUA_DIR}/src/lua51.lib)
endif(MSVC)

if(UNIX)
	set(LUA_LIB ${LUA_LIB} "dl")
endif(UNIX)


### V8 config

set(V8_DIR
	/source/v8
)

set(V8_INC ${V8_DIR}/include)

if(CMAKE_COMPILER_IS_GNUCC)
	set(V8_LIB ${V8_DIR}/libv8.a)
endif(CMAKE_COMPILER_IS_GNUCC)

if(MSVC)
	set(V8_LIB ${V8_DIR}/v8.lib)
endif(MSVC)



### glut config
set(GLUT_DIR
	/Source/freeglut
)

set(GLUT_INC ${GLUT_DIR}/include)

if(CMAKE_COMPILER_IS_GNUCC)
	set(GLUT_LIB ${GLUT_DIR}/src/libfreeglutdll.a opengl32 glu32)
endif(CMAKE_COMPILER_IS_GNUCC)

if(MSVC)
	set(GLUT_LIB ${GLUT_DIR}/src/freeglut.lib)
endif(MSVC)
